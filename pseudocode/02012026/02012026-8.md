8. Inside-Bar Breakout Strategy (Price Action Only)
   Concept: Identify inside bars (range within previous bar) and trade breakouts.

Indicators:

None besides OHLC; uses candle patterns.

timeframe = "5m"

populate_indicators(dataframe): # Identify inside bars
dataframe.is_inside_bar = (
(high < high.shift(1)) AND
(low > low.shift(1))
)
return dataframe

populate_entry_trend(dataframe):

    # LONG: break above mother bar high
    buy_signal = (
        (is_inside_bar) AND
        (close > high.shift(1))
    )

    # SHORT: break below mother bar low
    sell_signal = (
        (is_inside_bar) AND
        (close < low.shift(1))
    )

    dataframe.enter_long = buy_signal
    dataframe.enter_short = sell_signal
    return dataframe

populate_exit_trend(dataframe):

    # Simple RR-based exit
    exit_long = (
        (percentage_profit >= 1.0%) OR
        (percentage_loss <= -0.5%)
    )
    exit_short = (
        (percentage_profit >= 1.0%) OR
        (percentage_loss <= -0.5%)
    )

    dataframe.exit_long = exit_long
    dataframe.exit_short = exit_short
    return dataframe
